TS 自定义类型 #

可以直接导入的类型 #

type ExtractType<O, T> = { [K in keyof O]: O[K] extends T ? O[K] : unknown }

// 组合两个 object 的类型
type Diff<T extends string, U> = ({ [P in T]: P } &
{ [P in keyof U]: U[P] extends string ? string : never } & {
  [x: string]: never
})[T]

// 这里我们只以 string 类型的key作为示例
type ExtractStringKey<A> = Diff<Extract<keyof A, string>, ExtractType<A, string>>

type EventEmitter<T extends Record<string, unknown>> =
  MonoArgEmitter<T, OptionalKeys<T>> & BiArgEmitter<T, RequiredKeys<T>>

type AnyFunction<T> = (...args: any[]) => T

type PartialReturnType<T extends (...args: unknown[]) => unknown> = Partial<ReturnType<T>>

type RefElement = Nullable<HTMLElement>

type Nullable<T> = null | T

type CustomizedHTMLElement<T> = HTMLElement & T

type Indexable<T> = {
  [key: string]: T
}

type Hash<T> = Indexable<T>

type TimeoutHandle = number // ReturnType<typeof window.setTimeout>

type StyleValue = string | CSSProperties | Array<StyleValue>

type Mutable<T> = { -readonly [P in keyof T]: T[P] }

type MethodNames<T> = { [F in keyof T]: T[F] extends (...args: any) => any ? F: never }[keyof T]

type PropNames<T> = { [F in keyof T]: T[F] extends (...args: any) => any ? never: F }[keyof T]
